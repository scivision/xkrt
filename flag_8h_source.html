45typedef enum xkrt_task_flags_t
46{
47 TASK_FLAG_ZERO = 0,
48
49 // These flags changes the `task_t` data structure memory layout
50 // They must be set once for all before task instanciation
51
52 TASK_FLAG_ACCESSES = (1 << 0), // may have accesses
53 TASK_FLAG_DETACHABLE = (1 << 1), // completion is associated with the completion of events
54 TASK_FLAG_DEVICE = (1 << 2), // may execute on a different device than host
55 TASK_FLAG_DOMAIN = (1 << 3), // children tasks may have dependencies
56 TASK_FLAG_MOLDABLE = (1 << 4), // may be split
57 TASK_FLAG_GRAPH = (1 << 5), // children tasks must be recorded
58 TASK_FLAG_RECORD = (1 << 6), // has a record - which implies buffering of emitted command
59
60 // These flags are run-time indicators
61 // They can be dynamically set/unset, even after task creation
62
63 TASK_FLAG_GRAPH_RECORDING = (1 << 7), // currently recording a graph
64 TASK_FLAG_GRAPH_EXECUTE_COMMAND = (1 << 8), // recorded commands are also executed
65 TASK_FLAG_REQUEUE = (1 << 9), // must be re-queued after returning from its routine
66
67 // Use for debugging
68
69 TASK_FLAG_MAX = (1 << 10)
70
71 // support me in the future
72 // TASK_FLAG_CANCEL = (1 << X), // cancelled
73 // TASK_FLAG_UNDEFERED = (1 << Y), // suspend the current task execution until that task completed
74 // TASK_FLAG_PERSISTENT = (1 << Z), // persistence
75
76} xkrt_task_flags_t;
