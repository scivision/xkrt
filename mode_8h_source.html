41// To OpenMP dependencies
42//
43// omp modifier | (xkrt_access_mode_t, xkrt_access_concurrency_t)
44//
45// in = (ACCESS_MODE_R, _, )
46// out|inout = (ACCESS_MODE_W, ACCESS_CONCURRENCY_SEQUENTIAL)
47// mutexinoutset = (ACCESS_MODE_W, ACCESS_CONCURRENCY_COMMUTATIVE)
48// inoutset = (ACCESS_MODE_W, ACCESS_CONCURRENCY_CONCURRENT)
49//
50// scope is sort of always 'ACCESS_SCOPE_UNIFIED' as we cannot specify
51// dependency domains in 6.0
52
53/* access mode */
54typedef enum xkrt_access_mode_t
55{
56 ACCESS_MODE_R = 0b00000001, // read
57 ACCESS_MODE_W = 0b00000010, // write
58 ACCESS_MODE_RW = ACCESS_MODE_R | ACCESS_MODE_W,
59 ACCESS_MODE_V = 0b00000100, // incoherent access (= 'virtual' = dont really move the memory)
60 ACCESS_MODE_VW = ACCESS_MODE_W | ACCESS_MODE_V,
61 ACCESS_MODE_VR = ACCESS_MODE_R | ACCESS_MODE_V,
62 ACCESS_MODE_D = 0b00001000, // detached access = do not fulfill
63 // dependencies on task completion: the
64 // task is responsible of fulfillment
65 // the access itself
66} xkrt_access_mode_t;
67
68static inline const char *
69xkrt_access_mode_to_str(xkrt_access_mode_t mode)
70{
71 switch (mode)
72 {
73 case (ACCESS_MODE_V): return "ACCESS_MODE_V";
74 case (ACCESS_MODE_R): return "ACCESS_MODE_R";
75 case (ACCESS_MODE_W): return "ACCESS_MODE_W";
76 case (ACCESS_MODE_RW): return "ACCESS_MODE_RW";
77 default: return "ACCESS_MODE_UNKN";
78 }
79}
