46typedef enum xkrt_driver_type_t
47{
48 XKRT_DRIVER_TYPE_HOST = 0, // cpu driver
49 XKRT_DRIVER_TYPE_CUDA = 1, // cuda devices driver
50 XKRT_DRIVER_TYPE_ZE = 2, // level zero devices driver
51 XKRT_DRIVER_TYPE_CL = 3, // opencl driver
52 XKRT_DRIVER_TYPE_HIP = 4, // hip driver
53 XKRT_DRIVER_TYPE_SYCL = 5, // sycl driver
54 XKRT_DRIVER_TYPE_MAX = 6
55} xkrt_driver_type_t;
56
57typedef uint8_t xkrt_driver_type_bitfield_t;
58_Static_assert(XKRT_DRIVER_TYPE_MAX <= sizeof(xkrt_driver_type_bitfield_t)*8, "");
59
60/* get driver name by type */
61inline const char *
62xkrt_driver_name(xkrt_driver_type_t driver_type)
63{
64 switch (driver_type)
65 {
66 case (XKRT_DRIVER_TYPE_HOST): return "host";
67 case (XKRT_DRIVER_TYPE_CUDA): return "cuda";
68 case (XKRT_DRIVER_TYPE_HIP): return "hip";
69 case (XKRT_DRIVER_TYPE_ZE): return "ze";
70 case (XKRT_DRIVER_TYPE_CL): return "cl";
71 case (XKRT_DRIVER_TYPE_SYCL): return "sycl";
72 default: return "(null)";
73 }
74}
75
76/* get driver type by name */
77inline xkrt_driver_type_t
78xkrt_driver_type_from_name(const char * name)
79{
80 for (int i = 0 ; i < XKRT_DRIVER_TYPE_MAX ; ++i)
81 if (strcmp(name, xkrt_driver_name((xkrt_driver_type_t) i)) == 0)
82 return (xkrt_driver_type_t) i;
83 return XKRT_DRIVER_TYPE_MAX;
84}
85
86/* return true if the driver is supported by the XKRT build */
87inline int
88xkrt_driver_supported(xkrt_driver_type_t driver_type)
89{
90 switch (driver_type)
91 {
92 case (XKRT_DRIVER_TYPE_HOST): return 1;
93 case (XKRT_DRIVER_TYPE_CUDA): return XKRT_SUPPORT_CUDA;
94 case (XKRT_DRIVER_TYPE_HIP): return XKRT_SUPPORT_HIP;
95 case (XKRT_DRIVER_TYPE_ZE): return XKRT_SUPPORT_ZE;
96 case (XKRT_DRIVER_TYPE_CL): return XKRT_SUPPORT_CL;
97 case (XKRT_DRIVER_TYPE_SYCL): return XKRT_SUPPORT_SYCL;
98 default: return 0;
99 }
100}
