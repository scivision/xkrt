361 inline void foreach(
362 const uintptr_t p,
363 const size_t size,
364 int n,
365 const std::function<void(const int, const uintptr_t, const uintptr_t)> & func)
366 {
367 // compute number of commands to spawn
368 if (size < (size_t) n)
369 n = (int) size;
370
371 // compute chunk size
372 const size_t chunksize = size / n;
373 assert(chunksize > 0);
374
375 for (int i = 0; i < n; ++i) {
376 const uintptr_t a = p + i * chunksize;
377 const uintptr_t b = (i == n - 1) ? p + size : p + (i + 1) * chunksize;
378 func(i, a, b);
379 }
380 }

810 /* Allocate a new task and call its constructor */
811 template <bool use_thread_stack = true>
812 inline task_t * task_new(
813 const task_format_id_t fmtid,
814 task_flag_bitfield_t flags,
815 const void * args,
816 const size_t args_size,
817 const task_access_counter_t n_accesses
818 ) {
819 thread_t * thread = thread_t::get_tls();
820 assert(thread);
821
822 if (n_accesses)
823 assert(flags & TASK_FLAG_ACCESSES);
824
825 if (thread->current_task->flags & TASK_FLAG_GRAPH_RECORDING)
826 flags |= TASK_FLAG_RECORD;
827
828 const size_t task_size = task_compute_size(flags, n_accesses);
829 const size_t size = task_size + args_size;
830
831 task_t * task;
832
833 if (use_thread_stack)
834 {
835 // if the task stack is full
836 if (thread->memory_stack_ptr + size > thread->memory_stack_bottom + XKRT_THREAD_MAX_MEMORY)
837 {
838 LOGGER_INFO("Task stack full! Increase `XKRT_THREAD_MAX_MEMORY` and recompile to avoid this");
839
840 // wait for all tasks completion
841 this->task_wait();
842
843 // clear all tasks references
844 # if XKRT_SUPPORT_DEBUG
845 thread->tasks.clear();
846 # endif /* XKRT_SUPPORT_DEBUG */
847 this->reset_dependence_controllers();
848
849 // reset the stack
850 this->task_deallocate_all();
851 }
852
853 // else, get a new task
854 task = (task_t *) thread->memory_stack_ptr;
855 thread->memory_stack_ptr += size;
856 }
857 else
858 {
859 // TODO: this task is leaking atm
860 task = (task_t *) malloc(size);
861 }
862
863 # if XKRT_SUPPORT_DEBUG
864 thread->tasks.push_back(task);
865 # endif /* XKRT_SUPPORT_DEBUG */
866
867 new (task) task_t(fmtid, flags);
868 task->parent = thread->current_task;
869
870 if (flags & TASK_FLAG_RECORD)
871 {
872 task_rec_info_t * rec = TASK_REC_INFO(task);
873 new (rec) xkrt::task_rec_info_t();
874 }
875
876 if (args)
877 {
878 assert(args_size);
879 void * task_args = TASK_ARGS(task, task_size);
880 assert(task_args);
881 memcpy(task_args, args, args_size);
882 }
883
884 return task;
885 }

890 template <task_flag_bitfield_t flags>
891 inline task_t *
892 task_instanciate(
893 const device_unique_id_t device_unique_id,
894 const task_access_counter_t ac,
895 const task_accesses_setter_t & set_accesses,
896 const task_split_condition_t & split_condition,
897 const void * args,
898 const size_t args_size,
899 const task_format_id_t fmtid
900 ) {
901 assert( (flags & TASK_FLAG_DEVICE) || (device_unique_id == XKRT_UNSPECIFIED_DEVICE_UNIQUE_ID));
902 assert(!(flags & TASK_FLAG_ACCESSES) || (set_accesses && ac));
903 assert(!(flags & TASK_FLAG_MOLDABLE) || (split_condition && ac));
904
905 // retrieve tls
906 thread_t * tls = thread_t::get_tls();
907 assert(tls);
908
909 // create the task
910 task_t * task = this->task_new(fmtid, flags, args, args_size, ac);
911 assert(task);
912
913 if constexpr (flags & TASK_FLAG_DEVICE)
914 {
915 task_dev_info_t * dev = TASK_DEV_INFO(task);
916 new (dev) task_dev_info_t(device_unique_id, XKRT_UNSPECIFIED_TASK_ACCESS);
917 }
918
919 if constexpr (flags & TASK_FLAG_MOLDABLE)
920 {
921 task_mol_info_t * mol = TASK_MOL_INFO(task);
922 new (mol) task_mol_info_t(split_condition, args_size);
923 }
924
925 if constexpr (flags & TASK_FLAG_ACCESSES)
926 {
927 task_acs_info_t * acs = TASK_ACS_INFO(task);
928 new (acs) task_acs_info_t(ac);
929
930 access_t * accesses = TASK_ACCESSES(task);
931 assert(accesses);
932 assert(ac);
933 set_accesses(task, accesses);
934 assert(tls->current_task);
935 this->task_accesses_resolve(tls->current_task, accesses, ac);
936 }
937
938 # if XKRT_SUPPORT_DEBUG
939 snprintf(task->label, sizeof(task->label), "capture");
940 # endif
941
942 return task;
943 }

1074 inline void
1075 team_parallel_for(
1076 team_t * team,
1077 team_parallel_for_i_func_t func,
1078 const int up,
1079 const int low = 0,
1080 const int incr = 1
1081 ) {
1082 this->team_parallel_for(team, [=] (thread_t * thread) {
1083 int up_v = up;
1084 int low_v = low;
1085 int last_iter;
1086 team_t::parallel_for_thread_bounds(&last_iter, &low_v, &up_v, incr);
1087 if (incr > 0)
1088 {
1089 for (int i = low_v ; i <= up_v ; i += incr)
1090 func(thread, i);
1091 }
1092 else
1093 {
1094 LOGGER_FATAL("Not supported");
1095 }
1096
1097 }
1098 );
1099 }
