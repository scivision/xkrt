41typedef enum xkrt_access_concurrency_t
42{
43 ACCESS_CONCURRENCY_SEQUENTIAL = 0b00000001,
44 ACCESS_CONCURRENCY_COMMUTATIVE = 0b00000010,
45 ACCESS_CONCURRENCY_CONCURRENT = 0b00000100,
46} xkrt_access_concurrency_t;

48static inline const char *
49xkrt_access_concurrency_to_str(xkrt_access_concurrency_t concurrency)
50{
51 switch (concurrency)
52 {
53 case (ACCESS_CONCURRENCY_SEQUENTIAL): return "ACCESS_CONCURRENCY_SEQUENTIAL";
54 case (ACCESS_CONCURRENCY_COMMUTATIVE): return "ACCESS_CONCURRENCY_COMMUTATIVE";
55 case (ACCESS_CONCURRENCY_CONCURRENT): return "ACCESS_CONCURRENCY_CONCURRENT";
56 default: return "ACCESS_CONCURRENCY_UNKN";
57 }
58}
